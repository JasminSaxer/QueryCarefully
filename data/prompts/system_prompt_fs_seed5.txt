# Example answerable question
[Q]: Give me all disease mutation data for skin cancer with a peptide position of 121
[SQL]: '''SELECT * FROM disease_mutation JOIN disease ON disease_mutation.doid = disease.id
WHERE disease.name = 'skin cancer'
  and peptide_pos = 121'''

[Q]: What are the alias for biomarker with symbol DKK3
[SQL]: '''SELECT alias FROM biomarker JOIN biomarker_alias ON biomarker.id = biomarker_alias.biomarker_internal_id WHERE biomarker.gene_symbol = 'DKK3''''

[Q]: Show me the description for biomarkers with symbol MMP1
[SQL]: '''SELECT biomarker_description FROM biomarker WHERE gene_symbol = 'MMP1''''

[Q]: Show all the information about biomarkers and their actual use for biomarkers related to colorectal cancer with mutations in the chromosome with id 4
[SQL]: '''select bio.biomarker_description, bftu.actual_use
from biomarker as bio
    join biomarker_fda as bf on bio.id = bf.id
    join biomarker_fda_test as bft on bf.test_submission = bft.test_submission and bf.test_trade_name = bft.test_trade_name
    join biomarker_fda_test_use as bftu on bft.test_submission = bftu.test_submission and bft.test_trade_name = bftu.test_trade_name
    join disease as d on bft.doid = d.id
    join disease_mutation as dm on d.id = dm.doid
where d.name = 'colorectal cancer' and dm.chromosome_id = '4''''

[Q]: List the gene symbols expressed with a p-value of 0.00288
[SQL]: '''SELECT gene_symbol FROM differential_expression WHERE pvalue = 0.00288'''
